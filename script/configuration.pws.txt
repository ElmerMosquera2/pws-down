# ==========================================
# 1. INICIALIZACIÓN Y CACHÉ
# ==========================================
$global:PwsConfigFile = Join-Path $HOME ".pws_config.json"
$global:PwsDefaultFile = Join-Path $HOME ".pws_default.json"

# Estado de fábrica
$global:PwsState = @{
    Mode = "minimalista"
    Locals = $false
    Installs = $false
    AutoUpdate = $false
    Cache = @{ Python = $null; Cpp = $null }
}

# Cargar configuración actual si existe
if (Test-Path $global:PwsConfigFile) {
    try {
        $loaded = Get-Content $global:PwsConfigFile -Raw | ConvertFrom-Json -AsHashtable
        foreach ($key in $loaded.Keys) { $global:PwsState[$key] = $loaded[$key] }
    } catch {}
}

function Save-PwsConfig {
    $global:PwsState | ConvertTo-Json -Depth 3 | Set-Content $global:PwsConfigFile
}

# AutoUpdate al inicio
if ($global:PwsState.AutoUpdate) {
    Start-Job -ScriptBlock {
        $py = (python --version 2>&1) -replace "Python ", ""
        $cpp = (g++ --version 2>&1 | Select-Object -First 1) -match "(\d+\.\d+\.\d+)" | Out-Null; $cppVer = $matches[1]
        return @{ Python = $py; Cpp = $cppVer }
    } | Receive-Job -Wait -AutoRemoveJob -OutVariable jobResult | Out-Null
    
    if ($jobResult) {
        $global:PwsState.Cache.Python = $jobResult.Python
        $global:PwsState.Cache.Cpp = $jobResult.Cpp
        Save-PwsConfig
    }
}

# ==========================================
# 2. CONTROLADOR: pws (La interfaz de comandos)
# ==========================================
function pws {
    param(
        [switch]$u, [switch]$update,
        [switch]$a, [switch]$activate,
        [switch]$d, [switch]$disable,
        [switch]$c, [switch]$config,
        [Alias("actually--push")][switch]$ap,
        [Alias("default--push")][switch]$dp,
        [switch]$o, [switch]$ostentoso,
        [switch]$m, [switch]$minimalista,
        [switch]$l, [switch]$locals,
        [switch]$i, [switch]$installs,
        [switch]$AU, [Alias("AutoUpdate")][switch]$AutoUpdateFlag
    )

    $isUpdating = $false

    # --- Lógica de Actualización de Caché (-u) ---
    if ($u -or $update) {
        Write-Host "🔄 Escaneando versiones de software..." -ForegroundColor Cyan
        try {
            $pyVer = (python --version 2>&1) -replace "Python ", ""
            $global:PwsState.Cache.Python = $pyVer
            Write-Host "  ✅ Python: v$pyVer" -ForegroundColor Green
        } catch { $global:PwsState.Cache.Python = $null }

        try {
            g++ --version 2>&1 | Select-Object -First 1 -OutVariable cppRaw | Out-Null
            if ($cppRaw -match "(\d+\.\d+\.\d+)") {
                $global:PwsState.Cache.Cpp = $matches[1]
                Write-Host "  ✅ C++ (g++): v$($matches[1])" -ForegroundColor Green
            }
        } catch { $global:PwsState.Cache.Cpp = $null }
        $isUpdating = $true
    }

    # --- Lógica de Gestión de Configuración (-c) ---
    if ($c -or $config) {
        if ($ap) {
            # Guardar el estado actual como el nuevo predeterminado
            $global:PwsState | ConvertTo-Json -Depth 3 | Set-Content $global:PwsDefaultFile
            Write-Host "💾 Estado actual guardado como tu configuración PREDETERMINADA." -ForegroundColor Green
        }
        elseif ($dp) {
            # Cargar el estado predeterminado
            if (Test-Path $global:PwsDefaultFile) {
                $loaded = Get-Content $global:PwsDefaultFile -Raw | ConvertFrom-Json -AsHashtable
                foreach ($key in $loaded.Keys) { $global:PwsState[$key] = $loaded[$key] }
                Write-Host "♻️ Restaurado a tu configuración PREDETERMINADA guardada." -ForegroundColor Cyan
            } else {
                # Si no hay guardado, valores de fábrica
                $global:PwsState.Mode = "minimalista"
                $global:PwsState.Locals = $false
                $global:PwsState.Installs = $false
                $global:PwsState.AutoUpdate = $false
                Write-Host "♻️ Restaurado a los valores de FÁBRICA." -ForegroundColor DarkCyan
            }
            $isUpdating = $true
        }
    }

    # --- Lógica de Activación (-a) ---
    if ($a -or $activate) {
        if ($o -or $ostentoso) {
            $global:PwsState.Mode = "ostentoso"
            $global:PwsState.Locals = $true
            $global:PwsState.Installs = $true
            Write-Host "✨ Modo Ostentoso ACTIVADO." -ForegroundColor Magenta
        }
        elseif ($m -or $minimalista) {
            $global:PwsState.Mode = "minimalista"
            $global:PwsState.Locals = $false
            $global:PwsState.Installs = $false
            Write-Host "🧼 Modo Minimalista ACTIVADO." -ForegroundColor DarkGray
        }

        if ($l -or $locals) {
            $global:PwsState.Locals = $true
            $global:PwsState.Mode = "custom"
            Write-Host "⏱️  Módulos Locales ACTIVADOS." -ForegroundColor Yellow
        }
        if ($i -or $installs) {
            $global:PwsState.Installs = $true
            $global:PwsState.Mode = "custom"
            Write-Host "💻 Módulos de Software ACTIVADOS." -ForegroundColor Cyan
        }
        if ($AU -or $AutoUpdateFlag) {
            $global:PwsState.AutoUpdate = $true
            Write-Host "⚙️  Auto-Update de inicio ACTIVADO." -ForegroundColor Blue
        }
        $isUpdating = $true
    }

    # --- Lógica de Desactivación (-d) ---
    if ($d -or $disable) {
        if ($l -or $locals) {
            $global:PwsState.Locals = $false
            $global:PwsState.Mode = "custom"
            Write-Host "⏸️  Módulos Locales DESACTIVADOS." -ForegroundColor DarkGray
        }
        if ($i -or $installs) {
            $global:PwsState.Installs = $false
            $global:PwsState.Mode = "custom"
            Write-Host "⏸️  Módulos de Software DESACTIVADOS." -ForegroundColor DarkGray
        }
        if ($AU -or $AutoUpdateFlag) {
            $global:PwsState.AutoUpdate = $false
            Write-Host "⏸️  Auto-Update de inicio DESACTIVADO." -ForegroundColor DarkGray
        }
        $isUpdating = $true
    }

    if ($isUpdating) {
        Save-PwsConfig
    } elseif (-not ($u -or $update -or $a -or $activate -or $d -or $disable -or $c -or $config)) {
        Write-Host "Modo actual: $($global:PwsState.Mode)" -ForegroundColor Cyan
    }
}

# ==========================================
# 3. EL PROMPT (Latencia 0ms real)
# ==========================================
function prompt {
    $e = [char]27
    $promptString = ""

    # Módulos Locales (Hora y Tiempo)
    if ($global:PwsState.Mode -eq "ostentoso" -or $global:PwsState.Locals) {
        $now = [System.DateTime]::Now.ToString("HH:mm:ss")
        $promptString += "$e[90m[$now]$e[0m "
        
        $history = Get-History -Count 1 -ErrorAction Ignore
        if ($history) {
            $duration = ($history.EndExecutionTime - $history.StartExecutionTime).TotalMilliseconds
            if ($duration -gt 1000) {
                $sec = [math]::Round($duration / 1000, 2)
                $promptString += "$e[38;5;208m⏱️ ${sec}s$e[0m "
            } else {
                $duration = [math]::Round($duration)
                $promptString += "$e[33m⚡ ${duration}ms$e[0m "
            }
        }
    }

    # Módulos de Software
    if ($global:PwsState.Mode -eq "ostentoso" -or $global:PwsState.Installs) {
        if ($global:PwsState.Cache.Python) { $promptString += "$e[38;5;226m🐍 v$($global:PwsState.Cache.Python)$e[0m " }
        if ($global:PwsState.Cache.Cpp) { $promptString += "$e[38;5;33m⚙️ C++ v$($global:PwsState.Cache.Cpp)$e[0m " }
    }

    # Usuario y Ruta
    $path = $PWD.ProviderPath
    $shortPath = $path -replace "^$([regex]::Escape($HOME))", "~"
    $promptString += "$e[92m$env:USERNAME$e[0m+$e[96m$shortPath$e[0m"

    # Rama GIT
    $branch = $null
    $current = $path
    while ($current) {
        $headFile = "$current\.git\HEAD"
        if ([System.IO.File]::Exists($headFile)) {
            try {
                $headContent = [System.IO.File]::ReadAllLines($headFile)[0]
                if ($headContent -match "^ref: refs/heads/(.*)") { $branch = $matches[1] } 
                else { $branch = $headContent.Substring(0, [math]::Min($headContent.Length, 7)) }
            } catch {}
            break
        }
        $parent = [System.IO.Path]::GetDirectoryName($current)
        if ($parent -eq $current -or [string]::IsNullOrEmpty($parent)) { break }
        $current = $parent
    }

    if ($branch) {
        $promptString += " $e[93m($branch)$e[0m"
    }
    
    $host.UI.RawUI.WindowTitle = "PS: $shortPath"

    return "$promptString`n$e[95m>>$e[0m "
}
